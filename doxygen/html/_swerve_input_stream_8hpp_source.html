341 frc::ChassisSpeeds Get() {
342 auto& cfg = m_swerveDrive->GetConfig();
343
344 double maxLinear =
345 cfg.GetMaximumChassisLinearVelocity().value_or(m_maximumChassisLinearVelocity).value();
346 units::radians_per_second_t maxAngular =
347 cfg.GetMaximumChassisAngularVelocity().has_value()
348 ? units::radians_per_second_t{cfg.GetMaximumChassisAngularVelocity().value()}
349 : m_maximumChassisAngularVelocity;
350
351 frc::Translation2d scaledTranslation = ApplyTranslationScalar(
352 ApplyDeadband(m_controllerTranslationX()), ApplyDeadband(m_controllerTranslationY()));
353 scaledTranslation = ApplyTranslationCube(scaledTranslation);
354 scaledTranslation = ApplyAllianceAwareTranslation(scaledTranslation);
355
356 double vx = scaledTranslation.X().value() * maxLinear;
357 double vy = scaledTranslation.Y().value() * maxLinear;
358 double omega = 0.0;
359 frc::ChassisSpeeds speeds{};
360
361 SwerveInputMode newMode = FindMode();
362 if (m_currentMode != newMode) {
363 TransitionMode(newMode);
364 }
365
366 switch (newMode) {
367 case SwerveInputMode::TRANSLATION_ONLY: {
368 auto& pid = m_swerveDrive->GetConfig().GetRotationPID();
369 omega = pid.Calculate(units::radian_t{m_swerveDrive->GetGyroAngle()}.value(),
370 m_lockedHeading.value().Radians().value());
371 break;
372 }
373 case SwerveInputMode::ANGULAR_VELOCITY: {
374 omega = ApplyOmegaCube(ApplyRotationalScalar(ApplyDeadband(m_controllerOmega.value()()))) *
375 maxAngular.value();
376 break;
377 }
378 case SwerveInputMode::HEADING: {
379 auto& pid = m_swerveDrive->GetConfig().GetRotationPID();
380 double headingTarget =
381 std::atan2(m_controllerHeadingX.value()(), m_controllerHeadingY.value()());
382 omega =
383 pid.Calculate(units::radian_t{m_swerveDrive->GetGyroAngle()}.value(), headingTarget);
384 if (m_axisDeadband.has_value() &&
385 std::abs(m_controllerHeadingX.value()()) + std::abs(m_controllerHeadingY.value()()) <
386 m_axisDeadband.value()) {
387 omega = 0.0;
388 }
389 break;
390 }
391 case SwerveInputMode::AIM: {
392 auto& pid = m_swerveDrive->GetConfig().GetRotationPID();
393 auto currentHeading = frc::Rotation2d{units::radian_t{m_swerveDrive->GetGyroAngle()}};
394 auto relativeTrl = m_aimTarget.value()().RelativeTo(m_swerveDrive->GetPose()).Translation();
395 auto target = relativeTrl.Angle() + currentHeading;
396 omega = pid.Calculate(currentHeading.Radians().value(), target.Radians().value());
397 break;
398 }
399 }
400
401 m_currentMode = newMode;
402 speeds = frc::ChassisSpeeds{units::meters_per_second_t{vx}, units::meters_per_second_t{vy},
403 units::radians_per_second_t{omega}};
404 return ApplyTranslationHeadingOffset(ApplyRobotRelativeTranslation(speeds));
405 }

454 SwerveInputMode FindMode() {
455 if (m_translationOnlyEnabled.has_value() && m_translationOnlyEnabled.value()()) {
456 return SwerveInputMode::TRANSLATION_ONLY;
457 }
458 if (m_aimEnabled.has_value() && m_aimEnabled.value()()) {
459 if (m_aimTarget.has_value()) {
460 return SwerveInputMode::AIM;
461 }
462 std::cerr << "[YAMS SwerveInputStream] AIM mode enabled but no target set. "
463 "Call WithAim() first.\n";
464 }
465 if (m_headingEnabled.has_value() && m_headingEnabled.value()()) {
466 if (m_controllerHeadingX.has_value() && m_controllerHeadingY.has_value()) {
467 return SwerveInputMode::HEADING;
468 }
469 std::cerr << "[YAMS SwerveInputStream] HEADING mode enabled but no heading axes set. "
470 "Call WithControllerHeadingAxis() first.\n";
471 }
472 if (!m_controllerOmega.has_value()) {
473 std::cerr << "[YAMS SwerveInputStream] No rotation axis configured. "
474 "Call WithControllerRotationAxis() or WithControllerHeadingAxis(). "
475 "Falling back to TRANSLATION_ONLY.\n";
476 return SwerveInputMode::TRANSLATION_ONLY;
477 }
478 return SwerveInputMode::ANGULAR_VELOCITY;
479 }

481 void TransitionMode(SwerveInputMode newMode) {
482 // Exit current mode
483 switch (m_currentMode) {
484 case SwerveInputMode::TRANSLATION_ONLY:
485 m_lockedHeading = std::nullopt;
486 break;
487 case SwerveInputMode::HEADING:
488 case SwerveInputMode::AIM:
489 m_swerveDrive->ResetAzimuthPID();
490 break;
491 default:
492 break;
493 }
494 // Enter new mode
495 switch (newMode) {
496 case SwerveInputMode::TRANSLATION_ONLY:
497 m_lockedHeading = frc::Rotation2d{units::radian_t{m_swerveDrive->GetGyroAngle()}};
498 m_swerveDrive->ResetAzimuthPID();
499 break;
500 case SwerveInputMode::HEADING:
501 case SwerveInputMode::AIM:
502 m_swerveDrive->ResetAzimuthPID();
503 break;
504 default:
505 break;
506 }
507 }

547 frc::Translation2d ApplyAllianceAwareTranslation(frc::Translation2d translation) {
548 if (m_allianceRelative.has_value() && m_allianceRelative.value()()) {
549 if (m_robotRelative.has_value() && m_robotRelative.value()()) {
550 throw std::runtime_error{"Cannot use robot-oriented control with alliance-aware movement!"};
551 }
552 auto alliance = frc::DriverStation::GetAlliance();
553 if (alliance.has_value() && alliance.value() == frc::DriverStation::Alliance::kRed) {
554 return translation.RotateBy(frc::Rotation2d{units::degree_t{180.0}});
555 }
556 }
557 return translation;
558 }
559
560 frc::ChassisSpeeds ApplyTranslationHeadingOffset(frc::ChassisSpeeds speeds) {
561 if (m_translationHeadingOffsetEnabled.has_value() &&
562 m_translationHeadingOffsetEnabled.value()() && m_translationHeadingOffset.has_value()) {
563 frc::Translation2d vec{units::meter_t{speeds.vx.value()}, units::meter_t{speeds.vy.value()}};
564 auto rotated = vec.RotateBy(m_translationHeadingOffset.value());
565 return frc::ChassisSpeeds{units::meters_per_second_t{rotated.X().value()},
566 units::meters_per_second_t{rotated.Y().value()}, speeds.omega};
567 }
568 return speeds;
