134inline double CrtRatioFromStages(std::initializer_list<int> pairs) {
135 double r = 1.0;
136 const int* p = pairs.begin();
137 while (p + 1 < pairs.end()) {
138 r *= static_cast<double>(p[0]) / static_cast<double>(p[1]);
139 p += 2;
140 }
141 return r;
142}
