191 frc2::CommandPtr DriveToPose(frc::Pose2d pose) {
192 return frc2::cmd::RunOnce([this] {
193 ResetTranslationPID();
194 ResetAzimuthPID();
195 })
196 .AndThen(Drive([this, pose]() -> frc::ChassisSpeeds {
197 auto& azimuthPID = m_config.GetRotationPID();
198 auto& translationPID = m_config.GetTranslationPID();
199 auto currentPose = GetPose();
200 auto distance = GetDistanceFromPose(pose);
201 auto translationScalar = translationPID.Calculate(distance.value(), 0.0);
202 auto poseDiff = currentPose.RelativeTo(pose);
203 return frc::ChassisSpeeds::FromFieldRelativeSpeeds(
204 frc::ChassisSpeeds{
205 units::meters_per_second_t{poseDiff.X().value() * translationScalar},
206 units::meters_per_second_t{poseDiff.Y().value() * translationScalar},
207 units::radians_per_second_t{
208 azimuthPID.Calculate(currentPose.Rotation().Radians().value(),
209 pose.Rotation().Radians().value())}},
210 frc::Rotation2d{units::radian_t{GetGyroAngle()}});
211 }))
212 .WithName("Drive to Pose");
213 }

334 void UpdateTelemetry() {
335 UpdatePoseEstimator();
336 auto pose = GetPose();
337 auto currentStates = GetModuleStates();
338
339 m_gyroPublisher.Set(units::degree_t{GetGyroAngle()}.value());
340 m_desiredModuleStatesPublisher.Set(m_desiredModuleStates);
341 m_currentModuleStatesPublisher.Set(currentStates);
342 m_posePublisher.Set(pose);
343 m_desiredRobotRelChassisSpeedsPublisher.Set(m_desiredChassisSpeeds);
344 m_currentRobotRelChassisSpeedsPublisher.Set(GetRobotRelativeSpeed());
345 m_fieldRelChassisSpeedsPublisher.Set(GetFieldRelativeSpeed());
346
347 for (auto* mod : m_config.GetModules()) {
348 mod->UpdateTelemetry();
349 }
350
351 m_field2d.SetRobotPose(pose);
352 std::vector<frc::Pose2d> modulePoses;
353 modulePoses.reserve(NumModules);
354 for (size_t i = 0; i < NumModules; ++i) {
355 auto* mod = m_config.GetModules()[i];
356 auto location = *mod->GetConfig().GetLocation();
357 auto moduleTranslation = pose.Translation() + location.RotateBy(pose.Rotation());
358 auto moduleHeading = pose.Rotation() + currentStates[i].angle;
359 modulePoses.push_back(frc::Pose2d{moduleTranslation, moduleHeading});
360 }
361 m_field2d.GetObject("modules")->SetPoses(modulePoses);
362 }

370 void SimIterate() {
371 if (!m_simTimer.IsRunning()) m_simTimer.Start();
372 for (auto* mod : m_config.GetModules()) {
373 mod->SimIterate();
374 }
375 auto speeds = m_kinematics.ToChassisSpeeds(GetModuleStates());
376 m_simGyroAngle += units::degree_t{units::degrees_per_second_t{speeds.omega}.value() *
377 m_simTimer.Get().value()};
378 m_simTimer.Reset();
379 }

500 static frc::SwerveDriveKinematics<NumModules> BuildKinematics(const SwerveDriveConfig& config) {
501 assert(config.GetModules().size() == NumModules);
